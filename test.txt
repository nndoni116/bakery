"use strict";

$.ajax({
  url: 'news.json',
  dataType: 'json'
})
.done(function(data) {
  // 成功した時の処理
  const recentNews = data.slice(0, 3);
  const $listContainer = $('#news-list'); // jQueryのセレクタ

  recentNews.forEach(item => {
    $listContainer.append(`
      <li>
        <span class="date">${item.date}</span>
        <span class="category">${item.category}</span>
        <a href="${item.url}">${item.title}</a>
      </li>
    `);
  });
})
.fail(function() {
  // 失敗した時の処理
  console.error("データの読み込みに失敗しました");
});

(jquery)
$(function() {
  const $slides = $('.slide');
  let currentIndex = 0;

  // 最初の一枚を表示
  $slides.eq(currentIndex).addClass('active');

  // 5秒ごとに切り替える関数
  setInterval(function() {
    const nextIndex = (currentIndex + 1) % $slides.length;
    
    $slides.eq(currentIndex).removeClass('active');
    $slides.eq(nextIndex).addClass('active');
    
    currentIndex = nextIndex;
  }, 5000);
});


(素のjavascript)

document.addEventListener('DOMContentLoaded', () => {
  // 1. 全てのスライド（画像が入った箱）を取得する
  const slides = document.querySelectorAll('.slide');
  let currentIndex = 0; // 今何枚目を表示しているか（0から数える）

  // 2. 最初の一枚に「active」クラスをつけて表示させる
  if (slides.length > 0) {
    slides[currentIndex].classList.add('active');
  }

  // 3. 5秒ごとに実行するタイマーを作る
  setInterval(() => {
    // 現在表示されているスライドから「active」を消す
    slides[currentIndex].classList.remove('active');

    // 次の番号を計算する（3枚なら 0→1→2→0... とループさせる）
    currentIndex = (currentIndex + 1) % slides.length;

    // 新しい番号のスライドに「active」をつけて表示させる
    slides[currentIndex].classList.add('active');
  }, 5000); // 5000ミリ秒 = 5秒
});
